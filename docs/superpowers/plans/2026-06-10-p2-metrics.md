# P2 Metrics & Polling Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** MetricPoints flow into the partitioned store from (a) direct SQL polling of monitored Neon databases and (b) aggregation of stored Clerk events, on a 5-minute external scheduler, surfaced via a series API and dashboard sparklines.

**Architecture:** Per ARCHITECTURE.md §7 + spike deltas, adjusted for actual plan tiers: Vercel Hobby (no Obs Plus, daily-only native crons) and Neon Free (no OTel export, no consumption API). Therefore: **GitHub Actions** schedule (`*/5`) curls CRON_SECRET-guarded endpoints; **poll-neon** runs `pg_stat_activity` queries against monitored databases over the Neon serverless driver (WebSocket/443), **event-gated** (skip when org has no Event in the last 15 min) so monitored computes can still scale to zero; **poll-clerk** derives auth-rate metrics from our own stored events. Vercel runtime metrics are explicitly out (R1: plan-gated) — deploy events already arrive via webhooks.

**Tech Stack:** existing + `uplot` (D11) for sparklines. Neon connection strings sealed in `Connection.accessTokenEnc`.

**Key decisions:**
- D17: Scheduler = GitHub Actions (repo already on GitHub; Hobby crons daily-only). `vercel.ts` crons deferred until Pro.
- D18: Neon metrics v1 = `neon.connections.active/idle/total`, `neon.db.size_bytes` via SQL; per-database Service (`kind: neon.database`).
- D19: Event-gated polling: no recent org activity ⇒ skip SQL connect ⇒ monitored compute suspends normally (spike cost trap).
- D20: Metric writes = upsert on PK (serviceId, metric, ts) with 1-min bucket floor — idempotent, replay-safe (D15).

`packages/connectors/src/neon.ts`:
```ts
import { Client, neonConfig } from "@neondatabase/serverless";
import ws from "ws";

neonConfig.webSocketConstructor = ws;

export type NeonMetricSample = {
    metric: string;
    value: number;
};

export const NEON_METRICS_QUERY = `
SELECT
    count(*) FILTER (WHERE state = 'active')::int AS active,
    count(*) FILTER (WHERE state = 'idle')::int AS idle,
    count(*)::int AS total
FROM pg_stat_activity
WHERE backend_type = 'client backend'`;

export const NEON_DB_SIZE_QUERY = `SELECT pg_database_size(current_database())::bigint AS bytes`;

// Floor a timestamp to the start of its 1-minute bucket.
export function bucketMinute(ts: Date): Date {
    const floored = new Date(ts);
    floored.setSeconds(0, 0);
    return floored;
}

// Decide whether to poll: only when the org showed activity recently,
// so monitored computes can scale to zero (spike R2 cost trap).
export function shouldPoll(last_event_at: Date | null, now: Date, window_ms: number): boolean {
    if (!last_event_at) return false;
    return now.getTime() - last_event_at.getTime() <= window_ms;
}

export async function sampleNeonDatabase(connection_string: string): Promise<NeonMetricSample[]> {
    const client = new Client({ connectionString: connection_string });
    await client.connect();
    try {
        const conns = await client.query(NEON_METRICS_QUERY);
        const size = await client.query(NEON_DB_SIZE_QUERY);
        const row = conns.rows[0];
        return [
            { metric: "neon.connections.active", value: Number(row.active) },
            { metric: "neon.connections.idle", value: Number(row.idle) },
            { metric: "neon.connections.total", value: Number(row.total) },
            { metric: "neon.db.size_bytes", value: Number(size.rows[0].bytes) }
        ];
    } finally {
        await client.end();
    }
}
```

`app/api/cron/poll-neon/route.ts`:
```ts
import { unseal } from "@supercritical/crypto";
import {
    bucketMinute,
    sampleNeonDatabase,
    shouldPoll
} from "@supercritical/connectors";
import { prisma } from "@supercritical/db";
import { isAuthorizedCron } from "@/lib/cron-auth";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

const ACTIVITY_WINDOW_MS = 15 * 60 * 1000;

export async function GET(req: Request) {
    if (!isAuthorizedCron(req)) return new Response("unauthorized", { status: 401 });
    const master_key = process.env.MASTER_KEY;
    if (!master_key) return new Response("MASTER_KEY not configured", { status: 500 });

    const now = new Date();
    const ts = bucketMinute(now);
    const results: Record<string, string> = {};

    const connections = await prisma.connection.findMany({
        where: { provider: "NEON", status: "ACTIVE" }
    });

    for (const connection of connections) {
        if (!connection.accessTokenEnc) continue;
        const last_event = await prisma.event.findFirst({
            where: { orgId: connection.orgId },
            orderBy: { receivedAt: "desc" },
            select: { receivedAt: true }
        });
        if (!shouldPoll(last_event?.receivedAt ?? null, now, ACTIVITY_WINDOW_MS)) {
            results[connection.externalAccountId] = "skipped: idle";
            continue;
        }
        try {
            const connection_string = unseal(Buffer.from(connection.accessTokenEnc), master_key);
            const database = new URL(connection_string).pathname.slice(1) || "postgres";
            const samples = await sampleNeonDatabase(connection_string);
            const service = await prisma.service.upsert({
                where: {
                    connectionId_externalId: {
                        connectionId: connection.id,
                        externalId: `${connection.externalAccountId}/${database}`
                    }
                },
                create: {
                    orgId: connection.orgId,
                    connectionId: connection.id,
                    provider: "NEON",
                    kind: "neon.database",
                    externalId: `${connection.externalAccountId}/${database}`,
                    name: database
                },
                update: {}
            });
            for (const sample of samples) {
                await prisma.metricPoint.upsert({
                    where: {
                        serviceId_metric_ts: { serviceId: service.id, metric: sample.metric, ts }
                    },
                    create: {
                        orgId: connection.orgId,
                        serviceId: service.id,
                        metric: sample.metric,
                        ts,
                        value: sample.value
                    },
                    update: { value: sample.value }
                });
            }
            await prisma.pollCursor.upsert({
                where: { connectionId_key: { connectionId: connection.id, key: "neon.sql" } },
                create: {
                    connectionId: connection.id,
                    key: "neon.sql",
                    lastRunAt: now,
                    lastSuccessAt: now
                },
                update: { lastRunAt: now, lastSuccessAt: now, consecutiveFailures: 0 }
            });
            results[connection.externalAccountId] = `ok: ${samples.length} metrics`;
        } catch (err) {
            await prisma.pollCursor.upsert({
                where: { connectionId_key: { connectionId: connection.id, key: "neon.sql" } },
                create: {
                    connectionId: connection.id,
                    key: "neon.sql",
                    lastRunAt: now,
                    consecutiveFailures: 1
                },
                update: { lastRunAt: now, consecutiveFailures: { increment: 1 } }
            });
            results[connection.externalAccountId] = `error: ${err instanceof Error ? err.message : "unknown"}`;
        }
    }
    return Response.json({ ok: true, polled: results });
}
```

`app/api/cron/poll-clerk/route.ts` — aggregate stored clerk events into 5-min-bucket rates:
```ts
import { bucketMinute } from "@supercritical/connectors";
import { prisma } from "@supercritical/db";
import { isAuthorizedCron } from "@/lib/cron-auth";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

const WINDOW_MS = 5 * 60 * 1000;

const CLERK_RATE_METRICS: { metric: string; kinds: string[] }[] = [
    { metric: "clerk.signins.rate", kinds: ["clerk.session.created"] },
    { metric: "clerk.signups.rate", kinds: ["clerk.user.created"] }
];

export async function GET(req: Request) {
    if (!isAuthorizedCron(req)) return new Response("unauthorized", { status: 401 });
    const now = new Date();
    const ts = bucketMinute(now);
    const since = new Date(now.getTime() - WINDOW_MS);
    let written = 0;

    const clerk_services = await prisma.service.findMany({
        where: { provider: "CLERK" }
    });
    for (const service of clerk_services) {
        for (const { metric, kinds } of CLERK_RATE_METRICS) {
            const count = await prisma.event.count({
                where: { serviceId: service.id, kind: { in: kinds }, receivedAt: { gte: since } }
            });
            await prisma.metricPoint.upsert({
                where: { serviceId_metric_ts: { serviceId: service.id, metric, ts } },
                create: { orgId: service.orgId, serviceId: service.id, metric, ts, value: count },
                update: { value: count }
            });
            written += 1;
        }
    }
    return Response.json({ ok: true, services: clerk_services.length, metric_points: written });
}
```

`series/route.ts` — GET `?serviceId&metric&minutes=60`:
```ts
import { NextRequest } from "next/server";
import { prisma } from "@supercritical/db";
import { getOrgOrNull } from "@/lib/org";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

export async function GET(req: NextRequest) {
    const org = await getOrgOrNull();
    if (!org) return new Response("unauthorized", { status: 401 });
    const params = req.nextUrl.searchParams;
    const service_id = params.get("serviceId");
    const metric = params.get("metric");
    const minutes = Math.min(Number(params.get("minutes") ?? 60) || 60, 24 * 60);
    if (!service_id || !metric) return new Response("serviceId and metric required", { status: 400 });

    const since = new Date(Date.now() - minutes * 60 * 1000);
    const points = await prisma.metricPoint.findMany({
        where: { orgId: org.id, serviceId: service_id, metric, ts: { gte: since } },
        orderBy: { ts: "asc" },
        select: { ts: true, value: true }
    });
    return Response.json({
        points: points.map((p) => ({ t: p.ts.getTime(), v: p.value }))
    });
}
```

`components/charts/sparkline.tsx` — uPlot wrapper, 120×28px, amber line, no axes:
```tsx
"use client";

import { useEffect, useRef } from "react";
import uPlot from "uplot";
import "uplot/dist/uPlot.min.css";

export function Sparkline({ data }: { data: { t: number; v: number }[] }) {
    const root = useRef<HTMLDivElement>(null);
    const plot = useRef<uPlot | null>(null);

    useEffect(() => {
        if (!root.current || data.length === 0) return;
        const series: uPlot.AlignedData = [
            data.map((d) => d.t / 1000),
            data.map((d) => d.v)
        ];
        plot.current?.destroy();
        plot.current = new uPlot(
            {
                width: 120,
                height: 28,
                pxAlign: false,
                cursor: { show: false },
                legend: { show: false },
                scales: { x: { time: true } },
                axes: [{ show: false }, { show: false }],
                series: [{}, { stroke: "#ffb000", width: 1 }]
            },
            series,
            root.current
        );
        return () => plot.current?.destroy();
    }, [data]);

    if (data.length === 0) return <span className="text-terminal-dim">no data</span>;
    return <div ref={root} />;
}
```
